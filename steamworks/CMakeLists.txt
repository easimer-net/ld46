if(WIN32)
    if("${CMAKE_SIZEOF_VOID_P}" STREQUAL "4")
        set(STEAM_LIB_BASE "win32")
    else()
        set(STEAM_LIB_BASE "win64")
    endif()
    set(STEAMWORKS_LIB "${CMAKE_CURRENT_LIST_DIR}/${STEAM_LIB_BASE}/steam_api.lib" CACHE FILEPATH "Path to Steamworks Lib")
else()
    if("${CMAKE_SIZEOF_VOID_P}" STREQUAL "4")
        set(STEAM_LIB_BASE "linux32")
    else()
        set(STEAM_LIB_BASE "linux64")
    endif()
    set(STEAMWORKS_LIB "${CMAKE_CURRENT_LIST_DIR}/${STEAM_LIB_BASE}/libsteam_api.so" CACHE FILEPATH "Path to Steamworks Lib")
endif()
